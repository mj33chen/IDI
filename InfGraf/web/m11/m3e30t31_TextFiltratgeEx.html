Textures. Aplicació de textures planes. Tècniques de filtratge: Exemple d'opcions de filtratge 

   /* Especifiquem un objecte textura 2D */

   glBindTexture (GL_TEXTURE_1D, texname);

   /* Indiquem funcions de filtratge */

     /* si es produeix magnificació, un filtre lineal */
     glTexParameter (GL_TEXTURE2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);

     /* si es produeix minimització, el texel més proper */
     glTexParameter (GL_TEXTURE2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);

     /* si excedim el rang en la coordenada s de textura, repetir textura en
        aquesta coordenada */
     glTexParameter (GL_TEXTURE2D, GL_TEXTURE_WRAP_S, GL_REPEAT);

     /* si excedim el rang en la coordenada t de textura, trucar a valor
        extrem del rang */
     glTexParameter (GL_TEXTURE2D, GL_TEXTURE_WRAP_T, GL_CLAMP);

   /* Especificar la imatge de contindrà la textura */
    glTexImage2D (GL_TEXTURE2D, 0, GL_RGBA, 32, 32, 0, GL_RGBA, type,
                         image);

   /* Habilitar les textures, l'activa és el darrer objecte especificat */
   glEnable(GL_TEXTURE_2D);

   /* pintem l'objecte especificant o no les coordenades de textura segons
      tinguem assignació explícita o automàtica */
   PintaObjecte;
